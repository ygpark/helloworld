	# <Executable name>
	ExecuteMe
	# </Executable name>

	# <Additional sources>

# To support make distclean.
# ===========================================================================
IF (UNIX)
	ADD_CUSTOM_TARGET (distclean @echo cleaning for source distribution)
	SET(DISTCLEANED
		cmake_install.cmake
		CMakeCache.txt
		Makefile
	)
  
	ADD_CUSTOM_COMMAND(
		DEPENDS clean
		COMMENT "distribution clean"
		COMMAND rm
		ARGS    -Rf CMakeFiles ${DISTCLEANED}
		TARGET  distclean
	)
ENDIF(UNIX)
